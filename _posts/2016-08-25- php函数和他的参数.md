function sum() {
    $acc = 0;
    foreach (func_get_args() as $n) { //这个函数就是拿到参数的函数.
        $acc += $n;
    }
    return $acc;
}

$callback =
			function ($quantity, $product) use ($tax, &$total) //use关键字在这里
			{
				$pricePerItem = constant(__CLASS__ . "::PRICE_" .
					strtoupper($product));
				$total += ($pricePerItem * $quantity) * ($tax + 1.0);
			};
